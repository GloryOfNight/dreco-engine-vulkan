cmake_minimum_required(VERSION 3.16.2)
project(dreco-launcher)

set(SOURCES_LIST main.cxx)

add_executable(${PROJECT_NAME} ${SOURCES_LIST})

target_link_libraries(${PROJECT_NAME} dreco)

get_target_property(DRECO_BINARY_DIR dreco BINARY_DIR)

set_target_properties(${PROJECT_NAME} PROPERTIES
    CXX_STANDARD 17
    CXX_STANDARD_REQUIRED YES
    CXX_EXTENSIONS NO
    EXPORT_COMPILE_COMMANDS TRUE
    RUNTIME_OUTPUT_DIRECTORY ${DRECO_BINARY_DIR})

install(DIRECTORY ${DRECO_SHADERS_BINARY_DIR_GLOBAL} DESTINATION ${CMAKE_INSTALL_PREFIX}/shaders)
install(DIRECTORY ${DRECO_ASSETS_DIR_GLOBAL} DESTINATION ${CMAKE_INSTALL_PREFIX})

install(TARGETS dreco dreco-launcher
        RUNTIME DESTINATION ${CMAKE_INSTALL_PREFIX}
        LIBRARY DESTINATION ${CMAKE_INSTALL_PREFIX}
        ARCHIVE EXCLUDE_FROM_ALL)
